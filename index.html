function renderHome() { // 1. 獲取今天的日期與月份 const now = new Date(); const currentMonth = now.getMonth() + 1; // 1-12 const currentYear = now.getFullYear(); // 2. 過濾出「本月」的活動 const thisMonthEvents = allEvents.filter(ev => { const evDate = new Date(ev.date); // 檢查年份與月份是否同時符合 return (evDate.getMonth() + 1) === currentMonth && evDate.getFullYear() === currentYear; }); // 3. 渲染到「本月精選活動」區域 const featuredGrid = document.getElementById('featured-grid'); if (thisMonthEvents.length > 0) { // 如果有活動，顯示本月的前 6 個 featuredGrid.innerHTML = thisMonthEvents.slice(0, 6).map(ev => createCard(ev, 'grid')).join(''); } else { // 如果本月沒活動，顯示溫馨提示 featuredGrid.innerHTML = `

☕ 本月目前暫無精選活動

請切換至「日曆」或「列表」查看更多日期

`; } // --- 以下維持原本的邏輯 (有積分、線上活動等，不限月份) --- // 有積分活動 (讓醫師隨時能找學分) document.getElementById('credit-list').innerHTML = allEvents.filter(e => parseFloat(e.credit) > 0).slice(0, 4).map(ev => createCard(ev, 'row')).join(''); // 線上活動 document.getElementById('online-list').innerHTML = allEvents.filter(e => e.type.includes('線上')).slice(0, 4).map(ev => createCard(ev, 'row')).join(''); // 最新更新 (顯示最後加入試算表的 3 筆) document.getElementById('recent-updates').innerHTML = [...allEvents].reverse().slice(0, 3).map(ev => `
${ev.title} ${ev.date}
`).join(''); }
